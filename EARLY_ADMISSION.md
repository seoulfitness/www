# ADMISSION

```sql
CREATE TABLE IF NOT EXISTS seoulfitness_v2.EARLY_ADMISSIONS (
    EARLY_ADMISSION_ID INT AUTO_INCREMENT PRIMARY KEY COMMENT '수시(입시) 요강 아이디',
    ADMISSION_ID INT COMMENT '입시 요강 아이디',
    USE_CSAT_REFLECTED_SCORE VARCHAR(1) NOT NULL COMMENT '수능 반영시 여부 (Y/N)',
    USE_PHYSICAL_REFLECTED_SCORE VARCHAR(1) NOT NULL COMMENT '실기 반영 여부 (Y/N)',
    USE_INTERNAL_REFLECTED_SCORE VARCHAR(1) NOT NULL COMMENT '내신 반영 여부 (Y/N)',
    USE_INTERVIEW_REFLECTED_SCORE VARCHAR(1) NOT NULL COMMENT '면접 반영 여부 (Y/N)',
    CSAT_REFLECTED_SCORE DECIMAL(6, 2) COMMENT '수능 반영 점수(0.000 ~ 1000.000)',
    PHYSICAL_REFLECTED_SCORE DECIMAL(6, 2) COMMENT '실기 반영 점수(0.000 ~ 1000.000)',
    INTERNAL_REFLECTED_SCORE DECIMAL(6, 2) COMMENT '내신 반영 점수(0.000 ~ 1000.000)',
    INTERVIEW_REFLECTED_SCORE DECIMAL(6, 2) COMMENT '면접 반영 점수(0.000 ~ 1000.000)',
    ACCEPTED_COUNT VARCHAR(10) COMMENT '수시(입시) 모집 인원',
    EARLY_ADMISSION_MEMO TEXT COMMENT '수시(입시) 요강 메모',
    CREATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성일시',
    UPDATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일시',
    CREATED_BY VARCHAR(50) COMMENT '등록한 사람',
    UPDATED_BY VARCHAR(50) COMMENT '수정한 사람'
);
```
