# DISTRICTS

```sql
CREATE TABLE IF NOT EXISTS DISTRICTS (
    DISTRICT_ID INT AUTO_INCREMENT PRIMARY KEY COMMENT '구/군 아이디',
    PROVINCE_ID INT NOT NULL COMMENT '시/도 아이디',
    DISTRICT_NAME VARCHAR(50) NOT NULL COMMENT '구/군 이름',
    DISTRICT_MEMO TEXT COMMENT '구/군 메모',
    CREATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성일시',
    UPDATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일시',
    CREATED_BY VARCHAR(50) COMMENT '등록한 사람',
    UPDATED_BY VARCHAR(50) COMMENT '수정한 사람'
);
```

```sql
INSERT INTO DISTRICTS (DISTRICT_ID, DISTRICT_NAME, PROVINCE_ID)
SELECT id, name, province_id FROM seoulfitness.district;
```

```sql
UPDATE DISTRICTS SET CREATED_BY='sung2ne', UPDATED_BY='sung2ne';
```
