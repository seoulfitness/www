# HIGH_SCHOOLS

```sql
CREATE TABLE IF NOT EXISTS HIGH_SCHOOLS (
    HIGH_SCHOOL_ID INT AUTO_INCREMENT PRIMARY KEY COMMENT '고등학교 아이디',
    PROVINCE_ID INT NOT NULL COMMENT '시/도 아이디',
    DISTRICT_ID INT NOT NULL COMMENT '구/군 아이디',
    HIGH_SCHOOL_NAME VARCHAR(50) COMMENT '고등학교 이름',
    HIGH_SCHOOL_ADDRESS VARCHAR(255) COMMENT '고등학교 주소',
    HIGH_SCHOOL_PHONE VARCHAR(20) COMMENT '고등학교 전화번호',
    HIGH_SCHOOL_WEBSITE VARCHAR(255) COMMENT '고등학교 웹사이트',
    HIGH_SCHOOL_MEMO TEXT COMMENT '고등학교 메모',
    CREATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성일시',
    UPDATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일시',
    CREATED_BY VARCHAR(50) COMMENT '등록한 사람',
    UPDATED_BY VARCHAR(50) COMMENT '수정한 사람'
);
```
