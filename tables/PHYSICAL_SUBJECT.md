# CSAT_SUBJECTS

```sql
CREATE TABLE IF NOT EXISTS seoulfitness_v2.PHYSICAL_SUBJECTS (
    PHYSICAL_SUBJECT_ID INT AUTO_INCREMENT PRIMARY KEY COMMENT '실기 교과목 아이디',
    PHYSICAL_SUBJECT_NAME VARCHAR(50) NOT NULL COMMENT '실기 교과목 이름',
    PHYSICAL_SUBJECT_MEMO TEXT COMMENT '실기 교과목 메모',
    CREATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성일시',
    UPDATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일시',
    CREATED_BY VARCHAR(50) COMMENT '등록한 사람',
    UPDATED_BY VARCHAR(50) COMMENT '수정한 사람'
);

INSERT INTO seoulfitness_v2.PHYSICAL_SUBJECTS (PHYSICAL_SUBJECT_ID, PHYSICAL_SUBJECT_NAME, PHYSICAL_SUBJECT_MEMO)
SELECT id, name, memo FROM seoulfitness.section2_subject;

UPDATE seoulfitness_v2.PHYSICAL_SUBJECTS SET CREATED_BY='sung2ne', UPDATED_BY='sung2ne';
```
