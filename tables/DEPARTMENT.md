# DEPARTMENTS

```sql
CREATE TABLE IF NOT EXISTS seoulfitness_v2.DEPARTMENTS (
    DEPARTMENT_ID INT AUTO_INCREMENT PRIMARY KEY COMMENT '학과 아이디',
    DEPARTMENT_NAME VARCHAR(50) NOT NULL COMMENT '학과 이름',
    DEPARTMENT_ADDRESS VARCHAR(255) COMMENT '학과 주소',
    DEPARTMENT_PHONE VARCHAR(20) COMMENT '학과 전화번호',
    DEPARTMENT_URL VARCHAR(255) COMMENT '학과 웹사이트',
    DEPARTMENT_MEMO TEXT COMMENT '학과 메모',
    CREATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성일시',
    UPDATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일시',
    CREATED_BY VARCHAR(50) COMMENT '등록한 사람',
    UPDATED_BY VARCHAR(50) COMMENT '수정한 사람'
);
```

```sql
INSERT INTO seoulfitness_v2.DEPARTMENTS (DEPARTMENT_ID, DEPARTMENT_NAME)
```

```sql
UPDATE seoulfitness_v2.DEPARTMENTS SET CREATED_BY='sung2ne', UPDATED_BY='sung2ne';
```
