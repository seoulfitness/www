# PHYSICAL_RECORD_PERIOD (실기 기록 등록 기간)

```sql
-- 실기 기록 등록 기간
CREATE TABLE IF NOT EXISTS PHYSICAL_RECORD_PERIOD (
    CSAT_RECORD_PERIOD_ID INT AUTO_INCREMENT PRIMARY KEY COMMENT '실기 기록 등록 기간 아이디',
    START_DATE DATE NOT NULL COMMENT '기록 등록 시작일',
    END_DATE DATE NOT NULL COMMENT '기록 등록 종료일',
    TITLE VARCHAR(100) NOT NULL COMMENT '기록 등록 제목',
    MEMO TEXT COMMENT '기록 등록 메모',
    GRANT_BRANCH_USER CHAR(1) DEFAULT 'N' COMMENT '지점 회원의 기록 등록 권한 (Y : 있음, N : 없음)',
    GRANT_NORMAL_USER CHAR(1) DEFAULT 'N' COMMENT '일반 회원의 기록 등록 권한 (Y : 있음, N : 없음)',
    CREATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성일시',
    UPDATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일시',
    CREATED_BY VARCHAR(50) COMMENT '등록한 사람',
    UPDATED_BY VARCHAR(50) COMMENT '수정한 사람'
) COMMENT '실기 기록 등록 기간';
```
