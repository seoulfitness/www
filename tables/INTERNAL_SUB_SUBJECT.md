# INTERNAL_SUB_SUBJECT

```sql
-- 내신 세부 교과목
CREATE TABLE IF NOT EXISTS INTERNAL_SUB_SUBJECT (
    INTERNAL_SUB_SUBJECT_ID INT AUTO_INCREMENT PRIMARY KEY COMMENT '내신 세부 교과목 아이디',
    INTERNAL_SUBJECT_ID INT NOT NULL COMMENT '내신 교과목 아이디',
    INTERNAL_SUB_SUBJECT_NAME VARCHAR(100) NOT NULL COMMENT '내신 세부 교과목 이름',
    INTERNAL_SUB_SUBJECT_MEMO TEXT COMMENT '내신 세부 교과목 메모',
    CREATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성일시',
    UPDATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일시',
    CREATED_BY VARCHAR(50) COMMENT '등록한 사람',
    UPDATED_BY VARCHAR(50) COMMENT '수정한 사람'
);
```
