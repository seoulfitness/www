# PROVINCES

```sql
CREATE TABLE IF NOT EXISTS seoulfitness_v2.PROVINCES (
    PROVINCE_ID INT AUTO_INCREMENT PRIMARY KEY COMMENT '시/도 아이디',
    PROVINCE_NAME VARCHAR(50) NOT NULL COMMENT '시/도 이름',
    PROVINCE_MEMO TEXT COMMENT '시/도 메모',
    CREATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성일시',
    UPDATED_AT DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일시',
    CREATED_BY VARCHAR(50) COMMENT '등록한 사람',
    UPDATED_BY VARCHAR(50) COMMENT '수정한 사람'
);
```

```sql
INSERT INTO seoulfitness_v2.PROVINCES (PROVINCE_ID, PROVINCE_NAME)
SELECT id, name FROM seoulfitness.province;
```

```sql
UPDATE seoulfitness_v2.PROVINCES SET CREATED_BY='sung2ne', UPDATED_BY='sung2ne';
```
